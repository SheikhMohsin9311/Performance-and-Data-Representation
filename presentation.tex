% 5. METHODOLOGY I: ARCHITECTURAL DETERMINISM
\begin{frame}{Methodology I: Architectural Determinism}
    \vspace{1.5em}
    \begin{columns}[T]
        \column{0.45\textwidth}
        \citepoint{System Isolation}\\
        \subtle{Eliminating OS-Level Variance}\\
        \vspace{0.8em}
        {\footnotesize\begin{itemize}
            \item \emphasis{CPU Pinning}: Benchmarks pinned to an isolated core to prevent context-switch overhead.
            \item \emphasis{Performance Governor}: Frequency locked at max clock to eliminate P-state ramp-up noise.
            \item \emphasis{ASLR Disabled}: Ensuring deterministic virtual-to-physical address mapping.
        \end{itemize}}

        \column{0.45\textwidth}
        \citepoint{Hardware State}\\
        \subtle{Consistency Across Runs}\\
        \vspace{0.8em}
        {\footnotesize\begin{itemize}
            \item \emphasis{Cold/Warm Control}: 3 leading warmup passes to prime branch predictors and L1/L2 caches.
            \item \emphasis{C-State Suppression}: Throttling deep-sleep states to maintain constant pipeline readiness.
        \end{itemize}}
    \end{columns}
\end{frame}

% 10. SCIENTIFIC DISCLOSURE: LIMITATIONS
\begin{frame}{Scientific Disclosure: Limitations}
    \vspace{1.5em}
    \begin{columns}[T]
        \column{0.45\textwidth}
        \citepoint{Architectural Specificity}\\
        \subtle{Hardware-Dependent Results}\\
        \vspace{0.8em}
        {\footnotesize\begin{itemize}
            \item \emphasis{Microarchitecture}: Results are specific to the Intel/AMD cache hierarchy used during testing.
            \item \emphasis{Single Operation}: Benchmarks isolate \emphasis{Traversal} performance; read/write/delete trade-offs are not explored here.
        \end{itemize}}

        \column{0.45\textwidth}
        \citepoint{Methodological Scope}\\
        \subtle{Trade-offs in Modeling}\\
        \vspace{0.8em}
        {\footnotesize\begin{itemize}
            \item \emphasis{Allocation Bias}: Measurements exclude \emphasis{malloc} overhead; real-world fragmentation impacts are simplified.
            \item \emphasis{Stationary Workloads}: Does not model dynamic churn or cache-pollution from interleaved tasks.
        \end{itemize}}
    \end{columns}
\end{frame}
